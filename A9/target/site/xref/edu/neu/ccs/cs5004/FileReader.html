3   import java.io.IOException;
4   import java.nio.file.Files;
5   import java.nio.file.Paths;
6   import java.util.stream.Stream;
7   
8   /**
9    * Read files of Template or ContactFile,and parse each line inside the file.
10   */
11  public class FileReader {
12    /**
13     * Read files.
14     * @param path the path for the input file
15     * @param parseFile parse interface
16     * @throws IOException throw io exception when there is errors
17     */
18    public static void readFile(String path, IParser parseFile) throws IOException {
19      try (Stream<String> stream = Files.lines(Paths.get(path))) {
20        stream.forEach(parseFile::parse);
21      } catch (IOException e) {
22        throw new IOException("There is no such file or there is problems on input of the file!");
23      }
24    }
25  }
